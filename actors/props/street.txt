ACTOR HydrantShootable
{
	//$Category Props (Wolf3D)/Street
	//$Title Hydrant (shootable)
	//$Color 3
	Health 1
	Radius 10
	Height 42
	Mass 999999
	+DONTGIB
	+NOBLOOD
	+NOICEDEATH
	+NOTAUTOAIMED
	+SHOOTABLE
	+SOLID
	DeathSound "hydrant/death"
	States
	{
	Spawn:
		HYDR A -1
		Stop
	Death:
		HYDB A 4 A_Scream
		"####" BC 4
	Water:
		TNT1 A 0 A_PlaySound("hydrant/broken", CHAN_7, frandom(0.8, 1.0), TRUE, ATTN_NORM)
	Broken:
		HYDR B 0 A_CheckRange(768,"BrokenSilent")
		HYDR BCDE 4
		Loop
	BrokenSilent:
		HYDR B 0 A_StopSound(CHAN_7)
		HYDR BCDE 4
		Goto Water
	}
}
