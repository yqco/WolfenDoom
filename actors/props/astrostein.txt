ACTOR FutureCeilingLight : SwitchableDecoration //3d actor
{
	//$Category Props (Wolf3D)/Astrostein
	//$Title Ceiling Light (modern, switchable)
	//$Color 3
	Radius 16
	Height 0
	-SOLID
	+NOGRAVITY
	+SPAWNCEILING
	States
	{
	Active:
		MDLA A 0 A_ChangeFlag("DORMANT", FALSE)
	Spawn:
	  MDLA A 0 NoDelay
	ActiveLights:
		MDLA A 0 A_SpawnItemEx("VolumetricLight_ConeDown", 0, 0, ScaleX+ScaleY*8, 0, 0, 0, 0, SXF_SETMASTER | SXF_CLIENTSIDE)
	Activate:
		MDLA A 1
		Loop
	Inactive:
		MDLA A 0 A_ChangeFlag("DORMANT", TRUE)
		MDLA A 0 A_RemoveChildren(TRUE, RMVF_MISC)
		MDLA B -1
		Stop
	}
}

ACTOR DecorationShip : SwitchableDecoration //3d actor
{
	//$Category Props (Wolf3D)/Astrostein
	//$Title Space Ship (viper, activatable)
	//$Color 3
	DistanceCheck "scenelod"
	SpawnID 211
	Radius 8
	Height 6
	Speed 6
	Damage 3
	SeeSound "viper/flyby"
	Projectile
	-ACTIVATEIMPACT
	+ACTIVATEPCROSS
	+CLIENTSIDEONLY
	+DONTSPLASH
	+NOGRAVITY
	States
	{
	Spawn:
	Active:
		MDLA A 1
		MDLA A 0 A_SpawnItemEx("DecorationShip_Steam", -32, -5, -1)
		MDLA A 0 A_SpawnItemEx("DecorationShip_Steam", -32, 5, -1)
		Loop
	Inactive:
		MDLA A -1
		Stop
	Death:
		MDLA A 4
		Stop
	}
}
