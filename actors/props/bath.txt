ACTOR ToiletShootable //3d actor
{
	//$Category Props (Wolf3D)/Bath
	//$Title Toilet (shootable)
	//$Color 3
	DistanceCheck "scenelod"
	Health 1
	Radius 10
	Height 42
	Mass 999999
	+DONTGIB
	+NOBLOOD
	+NOICEDEATH
	+NOTAUTOAIMED
	+SHOOTABLE
	+SOLID
	DeathSound "toilet/death"
	States
	{
	Spawn:
		MDLA A -1
		Stop
	Death:
		MDLA A 2 A_Scream
		TOIL BC 2
	Broken:
		TOIL D 0 A_CheckRange(512,"BrokenSilent")
		TOIL D 2 A_PlaySound("toilet/broken", CHAN_AUTO, 1.0, TRUE, ATTN_STATIC)
		"####" EFG 2
		Loop
	BrokenSilent:
		TOIL D 0 A_StopSound(CHAN_AUTO)
		TOIL D 2
		"####" EFG 2
		Goto Broken
	}
}
